		VARIÁVEIS COMPOSTAS HOMOGÊNEAS MULTIDIMENSIONAIS
		
			ALGORITMO "SEMNOME"
			
			VAR 
			
				M: VETOR[1..3, 1..2] DE INTEIRO
				
				//3 DIMENSÕES DE LINHAS, 2 DIMENSÕES DE COLUNAS
				//EM UMA ANALOGIA DE MEMÓRIA EM CÉLULAS (DE PLANILHAS)
				//OU VAGAS, EM ESTANTES DE ARMÁRIOS
				//MULTIDIMENSIONAIS OU MATRIZES POIS
				//É NECESSÁRIO 2 VALORES PARA IDENTIFICAR UMA VARIÁVEL 
				
				//A DIMENSÃO DA MATRIZ É IDENTIFICADA PELO VALOR DE SUAS DIMENSÕES 
				//O EXEMPLO ACIMA É UMA MATRIZ 3 por 2. "MATEMATICAMENTE, 3X2"
				
				
				L, C: INTEIRO 
				//CONTADORES
			
			INICIO 
			
				//COMO COLOCAR VALORES
				
				M[1,2] <- 4
				
				//NA VARIÁVEL "M", LINHA 1 COLUNA 2, RECEBE VALOR: "4"
				
				M[2,2] <- 5
				
				//NA VARIÁVEL "M", LINHA 2 COLUNA 2, RECEBE VALOR: "5"
				
				M[3,1] <- 8
				
				//NA VARIÁVEL "M", LINHA 3 COLUNA 1, RECEBE VALOR: "8"
			
				
				
				
				//OUTRO MÉTODO PARA ADICIONAR VALORES EM MATRIZES 
				
					PARA L <- 1 ATE 3 FACA
					//ESTE "PARA" USA O VALOR DA DIMENSÃO DAS "LINHAS" DA MATRIZ 
					
							PARA C <- 1 ATE 2 FACA 
							//ESTE "PARA" USA O VALOR DA DIMENSÃO "COLUNAS" DA MATRIZ 
							
								LEIA(M[L,C])
							
							FIMPARA
