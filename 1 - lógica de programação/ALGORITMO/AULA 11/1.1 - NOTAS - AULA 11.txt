	AULA 11
	ESTRUTURA DE REPETIÇÃO COM VARIAVEL DE CONTROLE
	
	
	
	
	REVISÃO
	
		ENQUANTO
			// É RECOMENDADO QUE O CONTADOR DA EXPRESSÃO SEJA DECLARADO
			// COMO "INTEIRO"

			// ENQUANTO A "EXPRESSÃO" FOR VERDADEIRA
				
				ENQUANTO (EXPRESSÃO - VERDADEIRO OU FALSO (1/0) ) FACA
				
					BLOCO
				FIMENQUANTO
				
				AO EXECUTAR O BLOCO, E CHEGAR NA ÚLTIMA LINHA DO "FIMENQUANTO", ELE TESTA NOVAMENTE A EXPRESSÃO. 
				
		REPITA
		
			//COMPARANDO COM O COMANDO ENQUANTO, O ENQUANTO EXECUTA O TESTE LÓGICO NO INÍCIO DA ROTINA.
			//PARA SABER SE DEVE OU NÃO EXECUTAR A ROTINA
			
			//O COMANDO REPITA, EXECUTA A ROTINA
			//ENTÃO, 
			//EXECUTA O TESTE LÓGICO PARA SABER SE DEVE OU NÃO REPETÍ-LA. 
			
			//REPETIRÁ ATÉ QUE A EXPRESSÃO SE TORNE VERDADEIRA
			
				REPITA
					
					BLOCO
					
				ATE (EXPRESSÃO - VERDADEIRO OU FALSO (1/0)
				
		
		
		
	ESTRUTURA DE UMA REPETIÇÃO: 

		//COM ENQUANTO
		
		C <- 1
		
		ENQUANTO (C <= 10) FACA
			BLOCO
			
			C <- C + 1
		FIMENQUANTO
		
		
		
		
		//COM REPITA
		
		C <- 1
		REPITA
			BLOCO
			
			C <- C + 1
			
		ATE (C > 10)
		
	
	
	
	NOVA ESTRUTURA DE REPETIÇÃO: 
	
	QUANDO SE SABE EXATAMENTE O Nº DE VEZES QUE UMA ROTINA DEVE 
	SER EXECUTADA, TEMOS OUTRA ESTRUTURA
	
	//INFORMAÇÕES ENTRE PARÊNTESES SÃO SUBSTITUÍVEIS, 
	//FORA DOS PARÊNTESES SÃO PARTE DA FUNÇÃO. 
	
	//INFORMAÇÃO ENTRE COLCHETE "[PASSO (SALTO)]" PODE SER OMITIDA
	//O PROGRAMA CONSIDERA "PASSO (1)"
	
	//QUANDO USA-SE "PARA (VARIAVEL) :- (INICIO)" ATRIBUI VALOR
	//DE INÍCIO A VARIÁVEL, ALTERANDO-A SE FOR NECESSÁRIO*?
	
		PARA (VARIAVEL) <- (INICIO) ATE (FIM) [PASSO (SALTO)] FACA
		
			BLOCO
			
		FIMPARA
		
		
			EXPLICAÇÃO: 
			NO INICIO, "C <- 1" (C, RECEBE 1)
			APÓS, INICIA-SE O BLOCO. 
			APÓS EXECUTADO O BLOCO, O PROGRAMA AVALIA O PASSO
			QUE EXECUTARÁ UM INCREMENTO ATÉ: "ATE (FIM)"
			SOMANDO O PASSO, NA VARIÁVEL DE CONTROLE "C"
			SERÁ EXECUTADO A 10ª REPETIÇÃO NO BLOCO, CASO
			O CONTROLE "C" ESTEJA EM (FIM)
	
	//A FUNÇÃO "PARA" NÃO PODE SUBSTITUIR "ENQUANTO" OU "REPITA", CASO
	//O NÚMERO DE REPETIÇÕES SEJA DEFINIDA POR UMA ENTRADA
	//PELO USUÁRIO DO PROGRAMA
	//ISSO É UM JULGAMENTO DE BOA PRÁTICA DO PROFESSOR
	//HÁ MÉTODOS, PORÉM, NÃO RECOMENDADOS
	
	//QUANDO UMA ESTRUTURA "PARA", É TERMINADA
	//E DEPOIS, EXECUTADA NOVAMENTE, SUA CONTAGEM É REINICIADA. 
	V.AULA11@1930
	
		
	
	EXEMPLO: 
		PARA (C) <- (1) ATE (10) PASSO (1) FACA
			ESCREVAL (C)
			
		FIMPARA
